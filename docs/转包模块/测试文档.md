```java
package com.example.wms.service.impl;

import com.example.wms.dto.transfer.TransferRequest;
import com.example.wms.dto.transfer.TransferResultDTO;
import com.example.wms.entity.InboundKanbanLabel;
import com.example.wms.entity.InventoryStock;
import com.example.wms.repository.*;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.boot.test.context.SpringBootTest;
import org.springframework.test.context.ActiveProfiles;
import org.springframework.transaction.annotation.Transactional;

import static org.junit.jupiter.api.Assertions.*;

@SpringBootTest
@ActiveProfiles("test")
@Transactional
public class TransferServiceImplTest {

    @Autowired
    private TransferService transferService;

    @Autowired
    private InboundKanbanLabelRepository kanbanLabelRepository;

    @Autowired
    private InventoryStockRepository inventoryStockRepository;

    @Autowired
    private InboundOrderRepository inboundOrderRepository;

    @Autowired
    private OutboundOrderRepository outboundOrderRepository;

    private InboundKanbanLabel testKanban;
    private InventoryStock testStock;

    @BeforeEach
    public void setUp() {
        // 创建测试用看板标签
        testKanban = new InboundKanbanLabel();
        testKanban.setKanbanNo("TEST-KANBAN-" + System.currentTimeMillis());
        testKanban.setMaterialCode("MAT001");
        testKanban.setMaterialName("测试物料");
        testKanban.setSupplierCode("SUP001");
        testKanban.setSupplierName("测试供应商");
        testKanban.setWarehouseArea("A区-01");
        testKanban.setLabelQty(100);
        testKanban.setLabelStatus("已入库");
        testKanban.setSealed(false);
        testKanban = kanbanLabelRepository.save(testKanban);

        // 创建测试用库存记录
        testStock = new InventoryStock();
        testStock.setMaterialCode("MAT001");
        testStock.setMaterialName("测试物料");
        testStock.setSupplier("测试供应商");
        testStock.setWarehouseArea("A区-01");
        testStock.setOnHandQty(100);
        testStock = inventoryStockRepository.save(testStock);
    }

    @Test
    public void testExecuteTransfer_Success() {
        // 准备请求
        TransferRequest request = new TransferRequest();
        request.setSourceKanbanNo(testKanban.getKanbanNo());
        request.setTransferQty(30);
        // 不指定目标看板号，测试自动生成

        // 执行转包
        TransferResultDTO result = transferService.executeTransfer(request, "tester");

        // 验证结果
        assertNotNull(result);
        assertEquals(testKanban.getKanbanNo(), result.getSourceKanbanNo());
        assertEquals(70, result.getSourceRemainingQty());
        assertEquals(30, result.getTargetQty());
        assertNotNull(result.getTargetKanbanNo());
        assertTrue(result.getTargetKanbanNo().startsWith("T-"));

        // 验证源看板状态更新
        InboundKanbanLabel updatedSource = kanbanLabelRepository.findByKanbanNo(testKanban.getKanbanNo()).orElse(null);
        assertNotNull(updatedSource);
        assertEquals(70, updatedSource.getLabelQty());
        assertEquals("转包", updatedSource.getTransferStatus());

        // 验证库存扣减
        InventoryStock updatedStock = inventoryStockRepository.findByMaterialCodeAndSupplierAndWarehouseArea(
                "MAT001", "测试供应商", "A区-01"
        ).orElse(null);
        assertNotNull(updatedStock);
        assertEquals(70, updatedStock.getOnHandQty());

        // 验证出库单创建
        assertTrue(outboundOrderRepository.findAll().stream()
                .anyMatch(o -> o.getOutboundType().equals("转包出库")));

        // 验证入库单创建
        assertTrue(inboundOrderRepository.findAll().stream()
                .anyMatch(i -> i.getRemark() != null && i.getRemark().contains("转包")));
    }

    @Test
    public void testExecuteTransfer_FullTransfer() {
        TransferRequest request = new TransferRequest();
        request.setSourceKanbanNo(testKanban.getKanbanNo());
        request.setTransferQty(100);  // 全量转包

        TransferResultDTO result = transferService.executeTransfer(request, "tester");

        // 验证源看板状态为"已转包"
        InboundKanbanLabel updatedSource = kanbanLabelRepository.findByKanbanNo(testKanban.getKanbanNo()).orElse(null);
        assertNotNull(updatedSource);
        assertEquals("已转包", updatedSource.getTransferStatus());
    }

    @Test
    public void testExecuteTransfer_KanbanNotFound() {
        TransferRequest request = new TransferRequest();
        request.setSourceKanbanNo("NOT-EXIST-KANBAN");
        request.setTransferQty(10);

        assertThrows(EntityNotFoundException.class, () -> {
            transferService.executeTransfer(request, "tester");
        });
    }

    @Test
    public void testExecuteTransfer_KanbanNotReceived() {
        // 将看板状态改为未入库
        testKanban.setLabelStatus("待入库");
        kanbanLabelRepository.save(testKanban);

        TransferRequest request = new TransferRequest();
        request.setSourceKanbanNo(testKanban.getKanbanNo());
        request.setTransferQty(10);

        IllegalStateException exception = assertThrows(IllegalStateException.class, () -> {
            transferService.executeTransfer(request, "tester");
        });
        assertTrue(exception.getMessage().contains("只有已入库的看板才能转包"));
    }

    @Test
    public void testExecuteTransfer_KanbanSealed() {
        // 将看板设为封存
        testKanban.setSealed(true);
        kanbanLabelRepository.save(testKanban);

        TransferRequest request = new TransferRequest();
        request.setSourceKanbanNo(testKanban.getKanbanNo());
        request.setTransferQty(10);

        IllegalStateException exception = assertThrows(IllegalStateException.class, () -> {
            transferService.executeTransfer(request, "tester");
        });
        assertTrue(exception.getMessage().contains("已封存"));
    }

    @Test
    public void testExecuteTransfer_InvalidQuantity_Zero() {
        TransferRequest request = new TransferRequest();
        request.setSourceKanbanNo(testKanban.getKanbanNo());
        request.setTransferQty(0);

        assertThrows(IllegalArgumentException.class, () -> {
            transferService.executeTransfer(request, "tester");
        });
    }

    @Test
    public void testExecuteTransfer_InvalidQuantity_Exceed() {
        TransferRequest request = new TransferRequest();
        request.setSourceKanbanNo(testKanban.getKanbanNo());
        request.setTransferQty(200);  // 超过可用数量

        assertThrows(IllegalArgumentException.class, () -> {
            transferService.executeTransfer(request, "tester");
        });
    }
}
```
